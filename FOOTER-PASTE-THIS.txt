<div class="tistory-lb" id="tsLb"><img src="" alt="" id="tsLbImg"/></div>
<script>
document.addEventListener('DOMContentLoaded',function(){
  // 라이트박스
  var ov=document.getElementById('tsLb'),im=document.getElementById('tsLbImg');
  document.querySelectorAll('.entry-content img,.wp-block-post-content img').forEach(function(i){
    if(i.src&&i.src.indexOf('your-site.com')>-1){i.style.cursor='zoom-in';i.addEventListener('click',function(e){e.preventDefault();im.src=this.src;ov.classList.add('active')})}
  });
  document.querySelectorAll('a[data-lightbox]').forEach(function(a){
    a.addEventListener('click',function(e){e.preventDefault();im.src=this.href;ov.classList.add('active')})
  });
  ov.addEventListener('click',function(){ov.classList.remove('active')});

  // 사이드바
  var sidebar=document.getElementById('tsSidebar'),overlay=document.getElementById('tsSidebarOverlay');
  var menuBtn=document.getElementById('tsMenuBtn'),closeBtn=document.getElementById('tsSidebarClose');
  function openSidebar(){sidebar.classList.add('open');overlay.classList.add('open');document.body.style.overflow='hidden'}
  function closeSidebar(){sidebar.classList.remove('open');overlay.classList.remove('open');document.body.style.overflow=''}
  if(menuBtn)menuBtn.addEventListener('click',openSidebar);
  if(closeBtn)closeBtn.addEventListener('click',closeSidebar);
  if(overlay)overlay.addEventListener('click',closeSidebar);

  // 검색
  var searchBtn=document.getElementById('tsSearchBtn'),searchOv=document.getElementById('tsSearchOverlay');
  var searchClose=document.getElementById('tsSearchClose'),searchInput=document.getElementById('tsSearchInput');
  if(searchBtn)searchBtn.addEventListener('click',function(){searchOv.classList.add('open');searchInput.focus()});
  if(searchClose)searchClose.addEventListener('click',function(){searchOv.classList.remove('open')});

  // ESC 키
  document.addEventListener('keydown',function(e){
    if(e.key==='Escape'){ov.classList.remove('active');closeSidebar();searchOv.classList.remove('open')}
  });
});
</script>
